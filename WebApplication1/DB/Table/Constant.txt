CREATE TABLE Component
(
Id int IDENTITY(1,1) PRIMARY KEY,
Name varchar(255),
CreatedTimeStamp datetime2 NOT NULL CONSTRAINT DF_Component_CreateDate_GETDATE DEFAULT GETDATE(),
UpdatedTimeStamp datetime2
)
